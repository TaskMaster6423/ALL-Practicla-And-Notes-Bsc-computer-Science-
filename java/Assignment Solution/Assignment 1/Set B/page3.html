B] Q.3] write a java program to loads names and phone numbers from a text file where the data is organized as one line per record and each field in a record are separeted by a tab(\t).it takes name or phone numbers as input and prints the corresponding other value from hash table(use hash table)

import java.io.*;
import java.util.Scanner;
import java.util.Hashtable;
public class B3 {
public static void main(String[] args) throws Exception{
try{
File f=new File("B3.txt");
BufferedReader br=new BufferedReader(new InputStreamReader(System.in));
Hashtabletable=new Hashtable<>();
Scanner sc=new Scanner(System.in);
String line="";
while((line=br.readLine())!=null){
String[]parts=line.split(":");
String name=parts[0].trim();
String number=parts[1].trim();
if(!name.equals("")&&! number.equals("")){
table.put(name,number);
}
}
System.out.println("Enter name:");
String key=sc.nextLine();
if(table.containsKey(key)){
System.out.println(table.get(key));
br.close();
sc.close();
}
}catch(Exception e){
System.out.println(e);
}
}
}
